validate()
{
if [ $1 -ne 0 ];
then
        echo "$2 .... failure"
        exit 1
else
        echo "success"
fi
}
yum install -y
validate $? "updating yum"

wget -0 /etc/yum.repos.d/jenkins.repo \
    https://pkg.jenkins.io/redhat-stable/jenkins.repo
validate $? "adding jenkins repo"

rpm --import https://pkg.jenkins.io/redhat-stable/jenkins.io-2023.key
validate $? "Jenkins-CI to enable installation from the package"

yum upgrade
validate $? "updating yum"

amazon-linux-extras install java-openjdk11 -y
validate $? "Install Java"

yum install jenkins -y
validate $? "Install Jenkins"

systemctl enable jenkins
validate $? "Enable the Jenkins service to start at boot"

systemctl start jenkins
validate $? "Start Jenkins as a service"

systemctl status jenkins
validate $? "checking the status of jenkins"
